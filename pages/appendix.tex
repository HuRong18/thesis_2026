\section{定理~\ref{proposition1}的证明} \label{<appendix-proposition1>}
为了证明此定理，对于每个集合 $\Pi_i$，定义一个不带模运算的索引集如下
\begin{equation}
	\hat{\Pi}_i = \{ \alpha_{i-1} + 1, \cdots, \alpha_{i-1} + \lceil \eta_i n \rceil \}, \quad \forall i \in \mathcal{J}.
	\label{eq:hat_pi}
\end{equation}
不难验证集合 $\Pi_i$ 和集合 $\hat{\Pi}_i$ 之间的映射是一一对应的。
因此，要证明定理~\ref{proposition1}，只需证明对于每个 $1 \leq j \leq m$，这 $r$ 个索引 $\{j, j+m, \cdots, j+(r-1)m\}$ 被包含在由\eqref{eq:hat_pi}定义的 $r$ 个不同的集合 $\hat{\Pi}_i$ 中；原因是，经过模$m$运算，这些索引中的每一个都将被映射为索引 $j$。
为此，观察到
\begin{align}
	\{j, j+m, \cdots, j+(r-1)m\} &\subset \{1, \cdots, rm\} 
	\subset \bigcup_{i=1}^{|\mathcal{J}|} \hat{\Pi}_i, \quad \forall j \in [m], \label{eq:subset_rel} 
\end{align}
其中最后一个包含关系归因于条件\eqref{eq:prop_cond1}和\eqref{eq:prop_cond2}。
由于集合 $\hat{\Pi}_i$ 是互斥的，且每个集合 $\hat{\Pi}_i$ 最多包含 $m$ 个元素（因为 $\eta_i \leq \frac{H(X)}{\log_2 q} = \frac{m}{n}$），\eqref{eq:subset_rel}表明这 $r$ 个索引确实被包含在至少$r$ 个不同的集合 $\hat{\Pi}_i$ 中。证毕。





\section{BP译码算法}\label{appendix-BP}
该算法将从长度为 $m$ 的伴随式序列 $\{S_{X_\upsilon}^{(j)}\}_{j=1}^m$\upcite{1042242}中解码出长度为 $n$ 的序列 $\{X_\upsilon^{(j)}\}_{j=1}^n$。
具体而言，在相应的 Tanner 图中，变量节点由 $\mathbf{X}_\upsilon^{1:n}$ 的元素组成，并在下文中为方便表示记为 $\{x_i\}_{i=1}^n$；
同样，校验节点由 $\mathbf{S}_{X_\upsilon}^{1:m}$ 的元素组成，并记为 $\{s_j\}_{j=1}^m$。
对于每个符号 $a \in GF(q)$，设 $M_{i \to j}(a)$ 为从变量节点 $x_i$ 发往校验节点 $s_j$ 的消息，$M_{j \to i}(a)$ 为从校验节点 $s_j$ 发往变量节点 $x_i$ 的消息。
基于伴随式的 BP 解码算法涉及以下四个步骤\upcite{1042242, 4697499, 4259766}：

初始化（Initialization）: 根据 $\{X_\upsilon^{(j)}\}_{j=1}^n$ 的分布，设定
\begin{equation}
	M_{i \to j}(a) = \text{Pr}\{x_i = a\}, \quad \forall a \in GF(q).
	\label{eq:bp_init}
\end{equation}
初始化之后，以下两个步骤将交替执行。

水平步骤（Horizontal step）: 对于每个校验节点 $s_j$，计算发往每个连接的变量节点 $x_i$ 的消息如下
\begin{align}
	M_{j \to i}(a) = \sum_{\{a_k\}_{k \in N(j)}} &\delta(a_i = a \text{ and } \sum_{k \in N(j)} H_{jk} a_k = s_j) \nonumber \\
	&\cdot \prod_{k \in N(j) \backslash i} M_{k \to j}(a_k), \quad \forall a \in GF(q),
	\label{eq:bp_horizontal}
\end{align}
其中当事件 $E$ 为真（假）时，$\delta(E)$ 等于一（零），$H_{jk}$ 是奇偶校验矩阵 $\mathbf{H}^{n \times m}$ 的元素，$N(j) \backslash i$ 表示除 $x_i$ 以外连接到 $s_j$ 的变量节点集合。

垂直步骤（Vertical Step）：对于每个变量节点 $x_i$，计算发往每个连接的校验节点 $s_j$ 的消息如下
\begin{equation}
	M_{i \to j}(a) = \frac{1}{Z_{i \to j}} \text{Pr}\{x_i = a\} \prod_{k \in N(i) \backslash j} M_{k \to i}(a), \quad \forall a \in GF(q),
	\label{eq:bp_vertical}
\end{equation}
其中 $Z_{i \to j}$ 是一个归一化因子，确保概率之和为一，$N(i) \backslash j$ 表示除 $s_j$ 以外连接到 $x_i$ 的校验节点集合。

判决（Decision）: 做出解码判决如下
\begin{equation}
	\hat{X}_\upsilon^{(i)} = \arg \max_{a \in GF(q)} \text{Pr}\{x_i = a\} \prod_{j \in N(i)} M_{j \to i}(a), \quad \forall i \in [n].
	\label{eq:bp_decision}
\end{equation}
解码过程将重复进行，直到满足 \eqref{sx} 或达到最大迭代次数。

\section{推论~\ref{bcst}的推导}\label{<corollary-derivation>}
本节说明推论~\ref{bcst}如何由定理~\ref{<theorem-chapter-2>}在无损且无边信息条件下特化得到。

定理~\ref{<theorem-chapter-2>}给出了在有损、有边信息条件下，恢复信源线性函数的可达速率区域。其速率约束为
\begin{equation}
	\sum_{i\in\mathcal{T}}R_i>H(\hat{\mathbf{Y}}_{\mathbf{E}}|X,\hat{\mathbf{Y}}_{\mathbf{C}\mathbf{E}})-H(U(\mathcal{T})|Y(\mathcal{T})),\label{general-rate}
\end{equation}
其中$U_k$为定理~\ref{<theorem-chapter-2>}中测试信道$p(u_k|y_k)$的输出，$U(\mathcal{T})=\{U_k\}_{k\in\mathcal{T}}$，$Y(\mathcal{T})=\{Y_k\}_{k\in\mathcal{T}}$。下面逐步将其特化到本文的梯度编码场景。

首先，由于本文场景中解码端（移动设备）不具备额外的边信息，令$X=\emptyset$，此时式\eqref{general-rate}中的条件熵退化为
\begin{equation}
	H(\hat{\mathbf{Y}}_{\mathbf{E}}|X,\hat{\mathbf{Y}}_{\mathbf{C}\mathbf{E}})=H(\hat{\mathbf{Y}}_{\mathbf{E}}|\hat{\mathbf{Y}}_{\mathbf{C}\mathbf{E}}).
\end{equation}
其次，在无损恢复条件下，要求失真$d_j=0$（$\forall j$）。为满足该零失真约束，令测试信道为恒等映射，即$U_k=Y_k$（$\forall k\in\mathcal{K}$），此时
\begin{equation}
	H(U(\mathcal{T})|Y(\mathcal{T}))=H(Y(\mathcal{T})|Y(\mathcal{T}))=0,
\end{equation}
因此式\eqref{general-rate}简化为
\begin{equation}
	\sum_{i\in\mathcal{T}}R_i>H(\hat{\mathbf{Y}}_{\mathbf{E}}|\hat{\mathbf{Y}}_{\mathbf{C}\mathbf{E}}).\label{simplified-rate}
\end{equation}
进一步，将定理~\ref{<theorem-chapter-2>}中的信源$Y_k$对应为参与传输的第$k$个边缘节点的编码梯度元素$\tilde{g}_{\pi(k),\cdot}^{(t)}$，速率$R_k$对应为信源编码速率$\tilde{R}_{c,\pi(k)}$，恢复系数矩阵$\mathbf{A}$对应为解码系数向量$\mathbf{a}=[a_{\pi(1)},\cdots,a_{\pi(K)}]$。代入式\eqref{simplified-rate}后，定义
\begin{equation}
	\hat{g}_{\mathbf{E}}^{(t)}=\mathbf{E}[\tilde{g}_{\pi(1),\cdot}^{(t)},\cdots,\tilde{g}_{\pi(K),\cdot}^{(t)}]^T,\quad \hat{g}_{\mathbf{C}\mathbf{E}}^{(t)}=\mathbf{C}\mathbf{E}[\tilde{g}_{\pi(1),\cdot}^{(t)},\cdots,\tilde{g}_{\pi(K),\cdot}^{(t)}]^T,
\end{equation}
即可得到推论~\ref{bcst}中的速率区域
\begin{equation}
	\mathcal{R}_{lossless}=\bigcup_{\mathbf{E}}\bigcap_{\mathbf{C}}\bigcup_{\mathcal{S}}\bigcap_{\mathcal{T}}\left\{(\tilde{R}_{c,\pi(1)},\cdots,\tilde{R}_{c,\pi(K)})\in\mathbb{R}_+^K: \sum_{i\in\mathcal{T}}\tilde{R}_{c,\pi(i)}> H(\hat{g}_{\mathbf{E}}^{(t)}|\hat{g}_{\mathbf{C}\mathbf{E}}^{(t)})\right\},
\end{equation}
其中集合运算的约束条件(a)--(d)与定理~\ref{<theorem-chapter-2>}一致，仅将span($\mathbf{A}$)替换为span($\mathbf{a}$)。

\section{定理~\ref{<linear>}的证明}\label{<linear-proof>}
推论~\ref{bcst}中的可达速率区域$\mathcal{R}_{lossless}$对辅助矩阵$\mathbf{E}$取并集，因此选取特定的$\mathbf{E}$可以得到$\mathcal{R}_{lossless}$的可达内界。以下分别取$\mathbf{E}=\mathbf{I}_K$和$\mathbf{E}=\mathbf{a}$两个特殊值，推导出两个可达边界。

（1）完整恢复边界$\mathcal{R}_{sw}$：
取辅助矩阵$\mathbf{E}=\mathbf{I}_K$，即$K\times K$单位矩阵，此时$L_{\mathbf{E}}=K$，且
\begin{equation}
	\hat{g}_{\mathbf{E}}^{(t)}=\mathbf{I}_K[\tilde{g}_{\pi(1),\cdot}^{(t)},\cdots,\tilde{g}_{\pi(K),\cdot}^{(t)}]^T=\tilde{g}_{\mathcal{K},\cdot}^{(t)},
\end{equation}
即解码器的目标是完整恢复所有参与传输节点的编码梯度元素。

由推论~\ref{bcst}的约束条件(b)，$\mathbf{C}\in\mathbb{F}_q^{L_{\mathbf{C}}\times K}$为行满秩矩阵，$L_{\mathbf{C}}$从$0$到$K-1$取遍所有值。当$\mathbf{E}=\mathbf{I}_K$时，$\hat{g}_{\mathbf{C}\mathbf{E}}^{(t)}=\mathbf{C}\mathbf{I}_K\tilde{g}_{\mathcal{K},\cdot}^{(t)}=\mathbf{C}\tilde{g}_{\mathcal{K},\cdot}^{(t)}$，因此推论~\ref{bcst}中的条件熵为
\begin{equation}
	H(\hat{g}_{\mathbf{E}}^{(t)}|\hat{g}_{\mathbf{C}\mathbf{E}}^{(t)})=H(\tilde{g}_{\mathcal{K},\cdot}^{(t)}|\mathbf{C}\tilde{g}_{\mathcal{K},\cdot}^{(t)}).
\end{equation}

由约束条件(c)和(d)，$|\mathcal{S}|=|\mathcal{T}|=L_{\mathbf{E}}-L_{\mathbf{C}}=K-L_{\mathbf{C}}$。由于$\mathbf{E}=\mathbf{I}_K$，约束条件(d)中$\mathbf{E}(\mathcal{S})=\mathbf{I}_K(\mathcal{S})$，即单位矩阵中由$\mathcal{S}$索引的行。此时秩条件为
\begin{equation}
	\mathrm{rank}\left(\begin{bmatrix}
		\mathbf{I}_K(\mathcal{S})\\
		\mathbf{I}_K(\{1,\cdots,K\}\backslash\mathcal{T})
	\end{bmatrix}\right)=K,
\end{equation}
即$\mathcal{S}\cup(\{1,\cdots,K\}\backslash\mathcal{T})=\{1,\cdots,K\}$，等价于$\mathcal{T}\subseteq\mathcal{S}$。又因为$|\mathcal{S}|=|\mathcal{T}|=K-L_{\mathbf{C}}$，故$\mathcal{T}=\mathcal{S}$。

下面对$L_{\mathbf{C}}$的不同取值分别讨论：

当$L_{\mathbf{C}}=0$时，$\mathbf{C}$为空矩阵，$|\mathcal{T}|=K$，即$\mathcal{T}=\mathcal{K}$。条件熵退化为$H(\tilde{g}_{\mathcal{K},\cdot}^{(t)})$，对应的速率约束为
\begin{equation}
	\sum_{k\in\mathcal{K}}\tilde{R}_{c,k}>H(\tilde{g}_{\mathcal{K},\cdot}^{(t)}).
\end{equation}

当$L_{\mathbf{C}}=K-1$时，$\mathbf{C}\in\mathbb{F}_q^{(K-1)\times K}$为行满秩矩阵，$|\mathcal{T}|=1$，即$\mathcal{T}=\{k\}$（$k\in\{1,\cdots,K\}$）。此时$\mathbf{C}\tilde{g}_{\mathcal{K},\cdot}^{(t)}$为$K-1$个线性无关组合，条件熵$H(\tilde{g}_{\mathcal{K},\cdot}^{(t)}|\mathbf{C}\tilde{g}_{\mathcal{K},\cdot}^{(t)})$表示已知$K-1$个线性无关组合后的剩余不确定性。对所有满足约束条件(c)的$\mathbf{C}$取交集，对应的速率约束为
\begin{equation}
	\tilde{R}_{c,k}>H(\tilde{g}_{\pi(k),\cdot}^{(t)}|\{\tilde{g}_{n,\cdot}^{(t)}\}_{n\in\mathcal{K}\backslash\{k\}}),\quad\forall k\in\{1,\cdots,K\}.
\end{equation}

对于一般的$L_{\mathbf{C}}\in\{0,\cdots,K-1\}$，$|\mathcal{T}|=K-L_{\mathbf{C}}$。对于给定的$\mathcal{T}$，遍历所有满足约束条件(c)的行满秩矩阵$\mathbf{C}\in\mathbb{F}_q^{L_{\mathbf{C}}\times K}$取交集后，$\mathbf{C}\tilde{g}_{\mathcal{K},\cdot}^{(t)}$能够表达$\mathcal{T}$的补集$\mathcal{K}\backslash\mathcal{T}$中各分量的全部信息。因此条件熵等价为
\begin{equation}
	H(\tilde{g}_{\mathcal{K},\cdot}^{(t)}|\mathbf{C}\tilde{g}_{\mathcal{K},\cdot}^{(t)})=H(\{\tilde{g}_{k,\cdot}^{(t)}\}_{k\in\mathcal{T}}|\{\tilde{g}_{n,\cdot}^{(t)}\}_{n\in\mathcal{K}\backslash\mathcal{T}}).
\end{equation}
当$L_{\mathbf{C}}$从$0$到$K-1$取遍所有值，$\mathcal{T}$遍历$\mathcal{K}$的所有非空子集，所产生的速率约束与经典SW约束具有相同形式\upcite{lim2019towards}
\begin{align}
		\sum_{k\in\mathcal{T}}\tilde{R}_{c,k}>H\left(\{\tilde{g}_{k,\cdot}^{(t)}\}_{k\in\mathcal{T}}|\{\tilde{g}_{n,\cdot}^{(t)}\}_{n\in\mathcal{K}\backslash\mathcal{T}}\right),\quad\forall \mathcal{T}\subseteq \mathcal{K}.\label{sw-constraint}
\end{align}
记该区域为$\mathcal{R}_{sw}$。

（2）线性恢复边界$\mathcal{R}_{linear}$：
取辅助矩阵$\mathbf{E}=\mathbf{a}$，即解码系数行向量，大小为$1\times K$，此时解码器的目标直接为$\hat{\mathbf{g}}_{\mathbf{a}}^{(t)}=\sum_{k\in\mathcal{K}}a_k\tilde{\mathbf{g}}_k^{(t)}=\mathbf{g}^{(t)}$。因此$L_{\mathbf{E}}=1$。由于推论~\ref{bcst}中的约束条件(b)要求$0\leq L_{\mathbf{C}}<L_{\mathbf{E}}=1$，故$L_{\mathbf{C}}$只能取$0$，即$\mathbf{C}$为空矩阵。此时$|\mathcal{S}|=|\mathcal{T}|=L_{\mathbf{E}}-L_{\mathbf{C}}=1$。

由于$\mathcal{S}\subseteq\{1,\cdots,L_{\mathbf{E}}\}=\{1\}$，故$\mathcal{S}=\{1\}$。对于$\mathcal{T}$，由约束条件(d)要求
\begin{equation}
	\mathrm{rank}\left(\begin{bmatrix}
		\mathbf{E}(\mathcal{S})\\
		\mathbf{I}_K(\{1,\cdots,K\}\backslash\mathcal{T})
	\end{bmatrix}\right)=K.\label{full rank}
\end{equation}
其中$\mathbf{E}(\mathcal{S})=\mathbf{a}$为$1\times K$的非零行向量，$\mathbf{I}_K(\{1,\cdots,K\}\backslash\mathcal{T})$为$\mathbf{I}_K$中去掉$\mathcal{T}$对应行后的$(K-1)\times K$子矩阵。设$\mathcal{T}=\{k\}$，则$\mathbf{I}_K(\{1,\cdots,K\}\backslash\{k\})$张成除第$k$维外的所有维度。要使式\eqref{full rank}满秩，需要$\mathbf{a}$的第$k$个分量$a_{\pi(k)}\neq 0$。若某个$a_{\pi(k)}=0$，则节点$\pi(k)$不参与全梯度的恢复，可将其从参与传输的节点集合$\mathcal{K}$中移除。因此不失一般性，假设所有参与传输节点的解码系数均满足$a_{\pi(k)}\neq 0$，故上述秩条件对所有$k\in\{1,\cdots,K\}$均成立。

由于$\mathbf{C}$为空，条件熵退化为无条件熵。考虑单个符号位置，对每个$\mathcal{T}=\{k\}$，代入推论~\ref{bcst}的速率约束可得
\begin{align}
	\tilde{R}_{c,k}>H(\hat{g}_{\mathbf{a}}^{(t)})=H\left(\sum_{j\in\mathcal{K}}a_j\tilde{g}_{j,\cdot}^{(t)}\right)=H(g^{(t)}),\quad\forall k\in \mathcal{K}.
\end{align}
记该区域为$\mathcal{R}_{linear}$。

综上所述，$\mathcal{R}_{sw}$和$\mathcal{R}_{linear}$分别对应推论~\ref{bcst}中$\mathbf{E}$取$\mathbf{I}_K$和$\mathbf{a}$时的可达内界。由于$\mathcal{R}_{lossless}$对$\mathbf{E}$取并集，两个特殊取值对应的区域之并$\mathcal{R}_{sub}=\mathcal{R}_{sw}\cup\mathcal{R}_{linear}$亦为$\mathcal{R}_{lossless}$的可达内界。证毕。

\section{推论~\ref{condition}的证明}\label{<condition>}
考虑所有参与传输节点的信道条件相同的情况，即$h_{\pi(1)}=h_{\pi(2)}=\cdots=h_{\pi(K)}$。由于信道条件完全对称，优化问题$\mathcal{P}_{4.1}$的最优解具有对称结构，即各节点的最优信源编码速率相等，记为$\tilde{R}_{c,k}=\tilde{R}_c$，$\forall k\in\mathcal{K}$。此时传输时延$T_d$由各节点共同的信源编码速率$\tilde{R}_c$决定，因此比较两种策略下的最优$\tilde{R}_c$即可确定时延的优劣。

对于$\mathcal{R}_{sw}$区域，将等速率$\tilde{R}_{c,k}=\tilde{R}_c$代入SW约束\eqref{sw-constraint}，对于每个非空子集$\mathcal{T}\subseteq\mathcal{K}$，需满足
\begin{align}
	\tilde{R}_c> \frac{H(\{\tilde{g}_{k,\cdot}^{(t)}\}_{k\in\mathcal{T}}|\{\tilde{g}_{n,\cdot}^{(t)}\}_{n\in\mathcal{K}\backslash\mathcal{T}})}{|\mathcal{T}|}.\label{sw-equal-rate}
\end{align}
$\mathcal{R}_{sw}$区域下的最优速率为上式右端对所有$\mathcal{T}$取最大值，即
\begin{align}
	\tilde{R}_c^{sw}=\max_{\mathcal{T}\subseteq\mathcal{K}}\frac{H(\{\tilde{g}_{k,\cdot}^{(t)}\}_{k\in\mathcal{T}}|\{\tilde{g}_{n,\cdot}^{(t)}\}_{n\in\mathcal{K}\backslash\mathcal{T}})}{|\mathcal{T}|}.
\end{align}
特别地，取$\mathcal{T}=\mathcal{K}$，此时$\mathcal{K}\backslash\mathcal{T}=\emptyset$，条件熵退化为联合熵，式\eqref{sw-equal-rate}给出
\begin{align}
	\tilde{R}_c>\frac{H(\{\tilde{g}_{k,\cdot}^{(t)}\}_{k\in\mathcal{K}})}{K}.
\end{align}
由于$\mathcal{T}=\mathcal{K}$是所有约束中的一个，$\tilde{R}_c^{sw}$作为所有约束右端的最大值，必然满足
\begin{align}
	\tilde{R}_c^{sw}\geq\frac{H(\{\tilde{g}_{k,\cdot}^{(t)}\}_{k\in\mathcal{K}})}{K}.\label{sw-lower-bound}
\end{align}

对于$\mathcal{R}_{linear}$区域，每个节点的速率约束为$\tilde{R}_{c,k}>H(g^{(t)})$，因此最优速率为
\begin{align}
	\tilde{R}_c^{linear}=H(g^{(t)}).
\end{align}

当推论~\ref{condition}中的条件$H(\{\tilde{g}_{k,\cdot}^{(t)}\}_{k\in\mathcal{K}})>K\cdot H(g^{(t)})$成立时，结合式\eqref{sw-lower-bound}可得
\begin{align}
	\tilde{R}_c^{sw}\geq\frac{H(\{\tilde{g}_{k,\cdot}^{(t)}\}_{k\in\mathcal{K}})}{K}>H(g^{(t)})=\tilde{R}_c^{linear}.
\end{align}
因此$\tilde{R}_c^{sw}>\tilde{R}_c^{linear}$，即线性恢复策略的每节点信源编码速率严格低于SW策略。由于信道条件相同，更低的信源编码速率直接对应更低的传输时延。证毕。
