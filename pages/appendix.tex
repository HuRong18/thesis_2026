\chapter{面向完整重构的DSC辅助编码计算结果传输方法研究}

\section{定理\ref{proposition1}的证明} \label{<appendix-proposition1>}
为了证明此命题，对于每个集合 $\Pi_i$，定义一个不带模运算的索引集如下
\begin{equation}
	\hat{\Pi}_i = \{ \alpha_{i-1} + 1, \cdots, \alpha_{i-1} + \lceil \eta_i n \rceil \}, \quad \forall i \in \mathcal{J}.
	\label{eq:hat_pi}
\end{equation}
不难验证集合 $\Pi_i$ 和集合 $\hat{\Pi}_i$ 之间的映射是一一对应的。
因此，要证明定理\ref{proposition1}，只需证明对于每个 $1 \leq j \leq m$，这 $r$ 个索引 $\{j, j+m, \cdots, j+(r-1)m\}$ 被包含在由 (\ref{eq:hat_pi}) 定义的 $r$ 个不同的集合 $\hat{\Pi}_i$ 中；原因是，经过模运算，这些索引中的每一个都将被映射为索引 $j$。
为此，观察到
\begin{align}
	\{j, j+m, \cdots, j+(r-1)m\} &\subset \{1, \cdots, rm\} \label{eq:subset_rel} \\
	&\subset \bigcup_{i=1}^{|\mathcal{J}|} \hat{\Pi}_i, \quad \forall j \in [m], \nonumber
\end{align}
其中最后一个包含关系归因于 (\ref{eq:prop_cond1}) 和 (\ref{eq:prop_cond2})。
由于集合 $\hat{\Pi}_i$ 是互斥的，且每个集合 $\hat{\Pi}_i$ 最多包含 $m$ 个元素（因为 $\eta_i \leq \frac{H(X)}{\log_2 q} = \frac{m}{n}$），(\ref{eq:subset_rel}) 表明这 $r$ 个索引确实被包含在（至少）$r$ 个不同的集合 $\hat{\Pi}_i$ 中。





\section{BP译码算法}\label{appendix-BP}
该算法将从长度为 $m$ 的伴随式序列 $\{S_{X_v}^{(j)}\}_{j=1}^m$ \cite{1042242} 中解码出长度为 $n$ 的序列 $\{X_v^{(j)}\}_{j=1}^n$。
具体而言，在相应的 Tanner 图中，变量节点由 $\mathbf{X}_v^{1:n}$ 的元素组成，并在下文中为方便表示记为 $\{x_i\}_{i=1}^n$；
同样，校验节点由 $\mathbf{S}_{X,v}^{1:m}$ 的元素组成，并记为 $\{s_j\}_{j=1}^m$。
对于每个符号 $a \in GF(q)$，设 $M_{i \to j}(a)$ 为从变量节点 $x_i$ 发往校验节点 $s_j$ 的消息，$M_{j \to i}(a)$ 为从校验节点 $s_j$ 发往变量节点 $x_i$ 的消息。
基于伴随式的 BP 解码算法涉及以下四个步骤\cite{1042242, 4697499, 4259766}：

\textbf{Initialization (初始化):} 根据 $\{X_v^{(j)}\}_{j=1}^n$ 的分布，设定
\begin{equation}
	M_{i \to j}(a) = \text{Pr}\{x_i = a\}, \quad \forall a \in GF(q).
	\label{eq:bp_init}
\end{equation}

初始化之后，以下两个步骤将交替执行。

\textbf{Horizontal step (水平步骤):} 对于每个校验节点 $s_j$，计算发往每个连接的变量节点 $x_i$ 的消息如下
\begin{align}
	M_{j \to i}(a) = \sum_{\{a_k\}_{k \in N(j)}} &\delta(a_i = a \text{ and } \sum_{k \in N(j)} H_{jk} a_k = s_j) \nonumber \\
	&\cdot \prod_{k \in N(j) \setminus i} M_{k \to j}(a_k), \quad \forall a \in GF(q),
	\label{eq:bp_horizontal}
\end{align}
其中当事件 $E$ 为真（假）时，$\delta(E)$ 等于一（零），$H_{jk}$ 是奇偶校验矩阵 $\mathbf{H}^{n \times m}$ 的元素，$N(j) \setminus i$ 表示除 $x_i$ 以外连接到 $s_j$ 的变量节点集合。

\textbf{Vertical step (垂直步骤):} 对于每个变量节点 $x_i$，计算发往每个连接的校验节点 $s_j$ 的消息如下
\begin{equation}
	M_{i \to j}(a) = \frac{1}{Z_{i \to j}} \text{Pr}\{x_i = a\} \prod_{k \in N(i) \setminus j} M_{k \to i}(a), \quad \forall a \in GF(q),
	\label{eq:bp_vertical}
\end{equation}
其中 $Z_{i \to j}$ 是一个归一化因子，确保概率之和为一，$N(i) \setminus j$ 表示除 $s_j$ 以外连接到 $x_i$ 的校验节点集合。

\textbf{Decision (判决):} 做出解码判决如下
\begin{equation}
	\hat{X}_v^{(i)} = \arg \max_{a \in GF(q)} \text{Pr}\{x_i = a\} \prod_{j \in N(i)} M_{j \to i}(a), \quad \forall i \in [n].
	\label{eq:bp_decision}
\end{equation}
解码过程将重复进行，直到满足 (\ref{sx}) 或达到最大迭代次数。


\chapter{面向线性恢复的DSC辅助编码计算结果传输方法研究}

\section{定理\ref{<linear>}的证明}\label{<linear-proof>}
基于Lim\ref{lim2019towards}的定理1，在无损（$D=0, U=\mathbf{g}$且无边信息（$X=\emptyset$）的条件下进行推导。
\subsection{完整恢复边界}
取辅助矩阵$\mathbf{E}=\mathbf{I}$，即单位矩阵，大小为$r\times r$，这意味着解码器的目标是恢复$\tilde{\mathbf{Y}}_{\mathbf{I}}=[\mathbf{g}_{\pi(1)}^{(t)},\cdots,\mathbf{g}_{\pi(r)}^{(t)}]$，即恢复所有原始局部计算结果。
因此，$L_{\mathbf{E}}=r$，由于需要恢复所有信息，干扰矩阵$\mathbf{C}$必须为空，即$L_{\mathbf{C}}=0$，否则无法区分具体信源。集合$\mathcal{S}=[1:r]$，代入定义1后可以得到
\begin{equation}
	\sum_{i\in\mathcal{T}}\eta_i>H(\hat{\mathbf{Y}}_\mathbf{E})-H(U(\mathcal{T})|g(\mathcal{T})
\end{equation}
由于无损，$H(U(\mathcal{T})|g(\mathcal{T})=0$。$H(\hat{\mathbf{Y}}_\mathbf{E})$即为向量$[\mathbf{g}_{\pi(1)}^{(t)},\cdots,\mathbf{g}_{\pi(r)}^{(t)}]$。
当$\mathcal{T}$取遍所有子集时，这就是经典的Slepian-Wolf区域
\begin{align}
		\sum_{n\in\mathcal{T}}\eta_n>H(\tilde{\mathbf{g}}(\mathcal{T})|\tilde{\mathbf{g}}(\mathcal{T}^c)),~\forall \mathcal{T}\subseteq \mathcal{G},
\end{align}

取$\mathcal{T}=\mathcal{G}$，和速率下界为
\begin{align}
	\sum_{n\in\mathcal{T}}\eta_n>H(\tilde{\mathbf{g}}_1,\tilde{\mathbf{g}}_2,\cdots,\tilde{\mathbf{g}}_r)
\end{align}
记为区域$\mathcal{h}_{sw}$。

\subsection{线性恢复边界}
取辅助矩阵$\mathbf{E}=\mathbf{B}$，即目标全梯度系数行向量，大小为$1\times r$，这意味着解码器的目标直接为$\tilde{\mathbf{Y}}_{\mathbf{B}}=\sum_{n\in\mathcal{G}}b_n g_n=\mathbf{g}^{(t)}$。
因此，$L_{\mathbf{E}}=1$，$L_{\mathbf{C}}=0$，集合$\mathcal{S}$只能取$\{1\}$。根据Lim\ref{lim2019towards}的定理1中的约束条件d要求
	\begin{equation}
	\textrm{rank}\left(\left[\begin{matrix}
		\mathbf{E}(\mathcal{S})\\
		\textbf{\textrm{I}}_r([r] \backslash \mathcal{T})
	\end{matrix}\right]\right)=r.\label{full rank}
\end{equation}
由于$\mathbf{E}=\mathbf{B}$是一个非零行向量，要使得公式（\ref{full rank}）满秩，每一个节点都必须包含在$\mathcal{T}$中或者被$\mathbf{I}$覆盖。在不考虑时间分片的情况下，这意味着对于每一个独立的节点$i\in\mathcal{G}$，其压缩率约束由$\mathbf{E}$决定。对于每一个单独的节点$i$（令$\mathcal{T}=\{i\}$），为了满足秩约束，不等式退化为
\begin{align}
	\eta_i>H(\tilde{\mathbf{Y}}_{\mathbf{B}})=H(\mathbf{g}^{(t)}),~\forall i\in \mathcal{G}.
\end{align}
其和速率下界为
\begin{align}
	\sum_{i\in\mathcal{G}}\eta_i\geq r\cdot H(g^{(t)}),
\end{align}
记为区域$\mathcal{h}_{linear}$。

\section{定理\ref{condition}的证明}\label{<condition>}
令$\eta_{sw}^{sum}$和$\eta_{linear}^{sum}$分别表示在$\mathcal{h}_{sw}$和$\mathcal{h}_{linear}$区域内实现无损恢复所需的最小和速率。
根据$\mathcal{h}_{sw}$的定义，其最小和 速率由所有信源的联合熵决定
\begin{align}
	\eta_{sw}^{sum}=H(\tilde{\mathbf{g}}_1,\tilde{\mathbf{g}}_2,\cdots,\tilde{\mathbf{g}}_r)
\end{align}
由于梯度编码引入了$(s+1)$重的数据冗余，该联合熵通常由去重后的原始数据分块决定。

根据$\mathcal{h}_{linear}$的定义，每个节点必须独立发送足够恢复目标函数$g^{(t)}$的信息量，因此，其最小和速率为
\begin{align}
	\eta_{linear}^{sum}=\sum_{i\in\mathcal{G}}H(g^{(t)})=r\cdot H(g^{(t)}).
\end{align}
当寻求$\eta_{linear}^{sum}<\eta_{sw}^{sum}$的条件，即
\begin{align}
	r\cdot H(g^{(t)})<	H(\left\{\tilde{\mathbf{g}}_n\right\}_{n\in\mathcal{G}})
\end{align}
证毕。

\section{嵌套线性码编码方案}\label{<nested linear coding>}
考虑节点$k\in \mathcal{K}$分别以编码速率$R_k$和$\hat{R}_k$并使用两个索引$m_k\in[2^{nR_k}]$和$l_k\in[2^{n\hat{R}_k}]$。定义$R_{max}:=\max{R_1,\cdots,R_k},\hat{R}_{max}:=\max{\hat{R}_1,\cdots,\hat{R}_k}$。定义$\text{m}_k(m_k),m_k\in[2^{nR_k}]$为$m_k$的$q$进制展开，每当$R_k<R_{max}$时，将补零至长度为$\lceil nR_{max}/\log(q)\rceil$。为了便于理解，本文后续假设对于所有速率，$\frac{nR_k}{\log(q)}$和$\frac{n\hat{R}_k}{\log(q)}$均为整数。由于矢量形式的$\mathbf{m}_k$和标量形式的$m_k$一一映射，因此后文中$m_k$将被用来表示其$q$进制展开向量$\mathbf{m}_k(m_k)$。定义由下面的步骤生成的$K$个码本的集合定义为一个$(2^{nR_1}\cdots,2^{nR_k}, 2^{n\hat{R}_1},\cdots,2^{n\hat{R}_K},q,n)$嵌套线性码。

固定一个分布$\Pi_{k=1}^K p(u_k|x_k)$和重构函数$\hat{\mathbf{w}}_{\mathbf{A}_j}=\mathbf{A}_j[u_1,\cdots,u_k]^T$，使得期望失真函数满足$E[d_j(\mathbf{W}_{\mathbf{A}},\hat{\mathbf{W}}_{\mathbf{A}})]\leq \frac{D_j}{1+\epsilon^{''}},\forall j\in[J]$，假设参数满足$\epsilon^{''}>\epsilon^{'}>\epsilon$。
\begin{itemize}
	\item[(1)] 码本生成：固定一个有限域$\mathbb{F}_q$和一个参数$\epsilon^{'}\in(0,1)$。随机生成一个$\kappa\times n$的矩阵$\mathbf{G}\in \mathbb{F}_q^{\kappa\times n}$，以及抖动向量$d_k^n \in \mathbb{F}_q^n,\forall k\in[K]$，其中$\kappa=\frac{n(R_{max}+\hat{R}_{max})}{\log (q)}$，且$\mathbf{G}$和$d_k^n$的每个元素都是根据有限域$\mathbb{F}_q$上的均匀分布独立随机生成的。
	
	对于每一个信源$k\in\mathcal{K}$，通过下式生成一个线性码$\mathcal{C}_k$，即
	\begin{equation}
		u_k^n(m_k,l_k):=u_k^n(\text{m}_k,\text{l}_k)=[\text{m}_k,\text{l}_k]\mathbf{G}\oplus d_k^n,
	\end{equation}
	其中，索引$m_k$的范围是$[2^{nR_k}]$，索引$l_k$的范围是$[2^{n\hat{R}_k}]$。根据这种构造方式，每个码字都是独立同分布且均匀分布的，即每个符号都服从有限域上的均匀分布，并且码字之间是相互独立的。
	
	\item [(2)] 编码：对于每个信源$k\in \mathcal{K}$，寻找一个索引对$(m_k,l_k)\in [2^{nR_k}]\times [2^{n\hat{R}_k}]$，使得生成的码字$u_k^n (m_k,l_k)$与观测到的信源序列$y_k^n$是$\epsilon^{'}$-联合典型的，即
	\begin{equation}
		(u_k^n(m_k,l_k),x_k^n)\in \mathcal{T}_{\epsilon^{'}}^{(n)}(U_k,X_k).
	\end{equation}
	如果存在多个这样的索引对，则从中随机挑选一个。如果不存在这样的索引对，则从所有可能的索引对$[2^{nR_k}]\times [2^{n\hat{R}_k}]$中随机选择一个。节点$k$将发送索引$m_k$至中心节点。
	
	给定一个系数矩阵$\mathbf{A}\in \mathbb{F}_q^{J\times K}$，定义索引的线性组合为$\text{m}_\mathbf{A}(\text{m}_1,\cdots,\text{m}_K)=\mathbf{A}\begin{bmatrix}
	\text{m}_1\\\cdots \\\text{m}_K
	\end{bmatrix}$和$\text{l}_\mathbf{A}(\text{l}_1,\cdots,\text{l}_K)=\mathbf{A}\begin{bmatrix}
	\text{l}_1\\\cdots\\\text{l}_K
\end{bmatrix}$。本文后续将用$\text{m}_\mathbf{A}$和$\text{l}_\mathbf{A}$代替$\text{m}_\mathbf{A}(\text{m}_1,\cdots,\text{m}_K)$和$\text{l}_\mathbf{A}(\text{l}_1,\cdots,\text{l}_K)$。进一步可以得到，$m_{\mathbf{A}_j}=\oplus_{k=1}^K A_{jk}m_k$以及$l_{\mathbf{A}_j}=\oplus_{k=1}^K A_{jk}l_k$，其中$\mathbf{A}_j$表示矩阵$\mathbf{A}$的第$j$行。

\item[(3)]译码：设定参数$\epsilon^{'}<\epsilon$，中心节点在接收到发送的索引$(\text{m}_1,\cdots,\text{m}_K)$和侧信息$y^n$后，译码器将搜索一个唯一的索引元组$\hat{l}_{\mathbf{A}}$，使得所有的码字$u_k^n$与侧信息$y_n$满足$\epsilon$-联合典型性，即
\begin{equation}
	\left(u_1^n(\text{m}_1, \hat{\text{l}}_1),\cdots,u_K^n(\text{m}_K, \hat{\text{l}}_K)\right)\in \mathcal{T}_{\epsilon}^{(n)},
\end{equation}
	其中，$\hat{\text{l}}_k\in [2^{nR_k}], \forall k\in \mathcal{K}$。如果译码器找到了唯一的索引元组，则将$\hat{w}_\mathbf{A}^n(m_\mathbf{A},\hat{l}_\mathbf{A})=[m_{\mathbf{A}},\hat{l}_\mathbf{A}]\mathbf{G}\oplus d_{\mathbf{A}^m}$作为其估计值，其中$m_{\mathbf{A}}$是接收到的索引的线性组合。
	否则，如果不存在这样的索引元组，或者存在多个这样的元组，则译码失败。
\end{itemize}
